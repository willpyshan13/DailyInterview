class Solution {
    fun canPartition(nums: IntArray): Boolean {
        val n = nums.size
        if (n < 2) {
            return false
        }
        var sum = 0
        var maxNum = 0
        for (num in nums) {
            sum += num
            maxNum = Math.max(maxNum, num)
        }
        if (sum % 2 != 0) {
            return false
        }
        val target = sum / 2
        if (maxNum > target) {
            return false
        }
        val dp = Array(n) { BooleanArray(target + 1) }
        for (i in 0 until n) {
            dp[i][0] = true
        }
        dp[0][nums[0]] = true
        for (i in 1 until n) {
            val num = nums[i]
            for (j in 1..target) {
                if (j >= num) {
                    dp[i][j] = dp[i - 1][j] or dp[i - 1][j - num]
                } else {
                    dp[i][j] = dp[i - 1][j]
                }
            }
        }
        return dp[n - 1][target]
    }
}

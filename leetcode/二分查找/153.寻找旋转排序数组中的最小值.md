class Solution {
    fun findMin(nums: IntArray): Int {
        var l = 0
        var h = nums.size - 1
        while (l < h) {
            val m = l + (h - l) / 2
            if (nums[m] <= nums[h]) {
                h = m
            } else {
                l = m + 1
            }
        }
        return nums[l]
    }
}

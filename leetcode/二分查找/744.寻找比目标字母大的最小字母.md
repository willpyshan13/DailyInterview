class Solution {
    fun nextGreatestLetter(letters: CharArray, target: Char): Char {
        val n = letters.size
        var l = 0
        var h = n - 1
        while (l <= h) {
            val m = l + (h - l) / 2
            if (letters[m] <= target) {
                l = m + 1
            } else {
                h = m - 1
            }
        }
        return if (l < n) letters[l] else letters[0]
    }
}

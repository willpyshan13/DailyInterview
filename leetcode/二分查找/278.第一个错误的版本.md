/* The isBadVersion API is defined in the parent class VersionControl.
      def isBadVersion(version: Int): Boolean = {} */
class Solution: VersionControl() {
    override fun firstBadVersion(n: Int) : Int {
        var l = 1
        var h = n
        while (l < h) {
            val mid = l + (h - l) / 2
            if (isBadVersion(mid)) {
                h = mid
            } else {
                l = mid + 1
            }
        }
        return l
	}
}

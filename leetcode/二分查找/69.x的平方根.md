class Solution {
    fun mySqrt(x: Int): Int {
        if (x <= 1) {
            return x
        }
        var l = 1
        var h = x
        while (l <= h) {
            val mid = l + (h - l) / 2
            val sqrt = x / mid
            when {
                sqrt == mid -> {
                    return mid
                }
                mid > sqrt -> {
                    h = mid - 1
                }
                else -> {
                    l = mid + 1
                }
            }
        }
        return h
    }
}

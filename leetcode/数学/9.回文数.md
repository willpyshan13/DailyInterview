class Solution {
    fun isPalindrome(x: Int): Boolean {
        if (x<0||(x%10 ==0&&x!=0)) return false
        var temp = x
        var palindromeInt = 0
        while (temp>palindromeInt){
            palindromeInt = palindromeInt*10+temp%10
            temp /=10
        }
        return temp == palindromeInt||temp == palindromeInt/10
    }
}

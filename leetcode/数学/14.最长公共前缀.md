class Solution {
    fun longestCommonPrefix(strs: Array<String>): String {
        if (strs.isEmpty()) return ""
        for (i in strs[0].indices) {
            for (j in 1..strs.size-1) {
                if (i == strs[j].length||strs[0][i]!=strs[j][i]){
                    return strs[0].subSequence(0,i).toString()
                }
            }
        }
        return strs[0]
    }
}

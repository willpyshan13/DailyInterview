class Solution {
    fun subsets(nums: IntArray): List<List<Int>> {
        val res: MutableList<List<Int>> = ArrayList()
        for (i in 0 until (1 shl nums.size)) {
            val sub: MutableList<Int> = ArrayList()
            for (j in nums.indices) if (i shr j and 1 == 1) sub.add(nums[j])
            res.add(sub)
        }
        return res
    }
}

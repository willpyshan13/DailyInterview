class Solution {
    val stack = Stack<String>()
    fun isValid(s: String): Boolean {
        if (s.isEmpty()) return false
        for (i in s.indices){
            if (s[i].toString() == "{"){
                stack.push("}")
            }else if(s[i].toString() == "["){
                stack.push("]")
            }else if(s[i].toString() == "("){
                stack.push(")")
            }else if (stack.isEmpty()||s[i].toString()!=stack.pop()){
                return false
            }
        }
        return stack.isEmpty()
    }
}

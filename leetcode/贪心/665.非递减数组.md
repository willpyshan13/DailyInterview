class Solution {
    fun checkPossibility(nums: IntArray): Boolean {
        var cnt = 0
        var i = 1
        while (i < nums.size && cnt < 2) {
            if (nums[i] >= nums[i - 1]) {
                i++
                continue
            }
            cnt++
            if (i - 2 >= 0 && nums[i - 2] > nums[i]) {
                nums[i] = nums[i - 1]
            } else {
                nums[i - 1] = nums[i]
            }
            i++
        }

        return cnt <= 1
    }
}

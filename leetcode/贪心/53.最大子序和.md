class Solution {
    fun maxSubArray(nums: IntArray): Int {
        if (nums.isEmpty()) {
            return 0
        }
        var preSum = nums[0]
        var maxSum = preSum
        for (i in 1 until nums.size) {
            preSum = if (preSum > 0) preSum + nums[i] else nums[i]
            maxSum = maxSum.coerceAtLeast(preSum)
        }
        return maxSum
    }
}

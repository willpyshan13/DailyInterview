class Solution {
    fun maxProfit(prices: IntArray): Int {
        val n = prices.size
        if (n == 0) return 0
        var soFarMin = prices[0]
        var max = 0
        for (i in 1 until n) {
            if (soFarMin > prices[i]) soFarMin = prices[i] else max = max.coerceAtLeast(prices[i] - soFarMin)
        }
        return max
    }
}

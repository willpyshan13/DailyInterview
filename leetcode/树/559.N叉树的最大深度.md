/**
 * Definition for a Node.
 * class Node(var `val`: Int) {
 *     var children: List<Node?> = listOf()
 * }
 */

class Solution {
    fun maxDepth(root: Node?): Int {
        if (root==null) return 0
        var depth = 0
        for (i in root.children.indices){
            depth = Math.max(depth,maxDepth(root.children[i]))
        }
        return depth+1
    }
}
